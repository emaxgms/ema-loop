## Infrastruttura finale

```
~/ema-loop/
├── triage.sh              gh issue list → JSON
├── dev-run-fix.sh          worktree → agy → commit → PR → Firebase → task log
└── cron-worker.sh          triage + parallel dev-run-fix &

~/projects/<repo>/
├── .git
├── worktrees/fix/<TASK_ID>/    git worktree isolato
└── task/<TASK_ID>.json         log persistente
```

**Flusso:** `cron-worker.sh [repo...]` → `triage.sh` (gh API, 0 token) → `dev-run-fix.sh &` (parallelo per ogni issue) → agy implementa → commit → push → PR → Firebase Preview → task log.

## triage.sh

```bash
#!/usr/bin/env bash
set -euo pipefail
export PATH="$HOME/.npm-global/bin:$HOME/.local/bin:$PATH"
REPOS=("$@")
[ ${#REPOS[@]} -eq 0 ] && echo "[]" && exit 0

RESULTS='[]'
for REPO in "${REPOS[@]}"; do
  ISSUES=$(gh issue list --repo "$REPO" --label bug,enhancement --state open \
    --json number,title,body,labels,createdAt,url \
    --jq '[.[] | select(.assignees | length == 0) | select(.labels | any(.name | IN("wontfix","duplicate","ai:handled","ai:in_progress","ai:failed")) | not) | {repo: $repo, number, title, body, labels: [.labels[].name], url, created_at: .createdAt}]' --arg repo "$REPO" 2>/dev/null || echo '[]')
  RESULTS=$(python3 -c "import sys,json; r=json.loads('''$RESULTS'''); n=json.loads('''$ISSUES'''.replace(\"'\",\"\\\\'\")); r.extend(n); print(json.dumps(r))" 2>/dev/null || echo "$RESULTS")
done
echo "$RESULTS"
```

## dev-run-fix.sh

```bash
#!/usr/bin/env bash
set -euo pipefail
export PATH="$HOME/.npm-global/bin:$HOME/.local/bin:$PATH"
REPO="${1:?}"; DESCRIPTION="${2:?}"; BASE="${3:-main}"

LOCAL_NAME="$(basename "$REPO" .git)"; PROJ="$HOME/projects/$LOCAL_NAME"
mkdir -p "$PROJ"; cd "$PROJ"; [ -d ".git" ] || gh repo clone "$REPO" "$PROJ"

TASK_ID=$(date +%Y%m%d-%H%M%S); TASK_DIR="$PROJ/task"; mkdir -p "$TASK_DIR"
TASK_FILE="$TASK_DIR/$TASK_ID.json"; BRANCH="fix/$TASK_ID"; WORKTREE="$PROJ/worktrees/$BRANCH"

cat > "$TASK_FILE" <<ENDJSON
{ "task_id": "$TASK_ID", "repo": "$REPO", "base": "$BASE",
  "description": $(echo "$DESCRIPTION" | python3 -c "import sys,json; print(json.dumps(sys.stdin.read()))"),
  "started_at": "$(date -u +%Y-%m-%dT%H:%M:%SZ)", "status": "started" }
ENDJSON

git fetch origin "$BASE"
git worktree add -b "$BRANCH" "$WORKTREE" "origin/$BASE" 2>/dev/null
cd "$WORKTREE"

agy -p "Implement the fix described below. Only edit files that are relevant. Commit with conventional commit.
---
$DESCRIPTION" --dangerously-skip-permissions 2>&1 || true

[ "$(git rev-parse HEAD)" = "$(git rev-parse origin/$BASE)" ] && {
  python3 -c "import json,datetime; d=json.load(open('$TASK_FILE')); d.update({'status':'no_changes','finished_at':datetime.datetime.utcnow().strftime('%Y-%m-%dT%H:%M:%SZ')}); json.dump(d,open('$TASK_FILE','w'),indent=2)"
  git worktree remove "$WORKTREE" 2>/dev/null || true; exit 0
}

CHANGES=$(git diff "origin/$BASE" --stat --diff-filter=AM 2>/dev/null | tail -n +1 | head -20 || echo "")
COMMIT_MSG=$(git log "origin/$BASE..HEAD" --format="%s" 2>/dev/null | head -5 || echo "")

git push -u origin "$BRANCH" 2>/dev/null || git push -u origin "$BRANCH" --force
PR_URL=$(gh pr create --repo "$REPO" --head "$BRANCH" --base "$BASE" \
  --title "$(git log -1 --format=%s 2>/dev/null || echo 'fix: automated fix')" \
  --body "$DESCRIPTION\n---\n_Generated by dev-runner_" 2>&1)
PR_NUMBER=$(echo "$PR_URL" | grep -oE '[0-9]+$' || echo "")

PREVIEW_URL=""
command -v firebase &>/dev/null && [ -f "$PROJ/firebase.json" ] && {
  FO=$(cd "$PROJ" && firebase hosting:channel:deploy "pr-${PR_NUMBER}" --json 2>/dev/null || true)
  PREVIEW_URL=$(echo "$FO" | python3 -c "import sys,json
try:
    d=json.load(sys.stdin)
    for v in d.get('result',{}).values():
        if isinstance(v,dict) and v.get('url'): print(v['ur'l]); break
except: pass" 2>/dev/null || echo "")
  [ -n "$PREVIEW_URL" ] && gh pr comment "$PR_NUMBER" --repo "$REPO" --body "🔗 **Firebase Hosting Preview:** [${PREVIEW_URL}](${PREVIEW_URL})"
}

git worktree remove "$WORKTREE" 2>/dev/null || true
echo "PR: $PR_URL"

python3 -c "
import json, datetime
with open('$TASK_FILE') as f: d = json.load(f)
d.update({'status':'done','pr_url':'$PR_URL','pr_number':${PR_NUMBER:-null},
  'preview_url':'${PREVIEW_URL:-}' or None,'commit_sha':'$(git -C "$PROJ" rev-parse "$BRANCH" 2>/dev/null || echo "")',
  'branch':'$BRANCH','changes':[] if not '$CHANGES' else '''$CHANGES'''.strip().split(chr(10)),
  'commit_msg':'$(echo "$COMMIT_MSG" | sed "s/'//g")',
  'finished_at':datetime.datetime.utcnow().strftime('%Y-%m-%dT%H:%M:%SZ')})
json.dump(d,open('$TASK_FILE','w'),indent=2)
print('Task: $TASK_FILE')
"
```

## cron-worker.sh

```bash
#!/usr/bin/env bash
set -euo pipefail
export PATH="$HOME/.npm-global/bin:$HOME/.local/bin:$PATH"
REPOS=("$@"); [ ${#REPOS[@]} -eq 0 ] && REPOS=("emaxgms/Antas-ai-news-feed")

ISSUES=$(bash "$(dirname "$0")/triage.sh" "${REPOS[@]}")
COUNT=$(echo "$ISSUES" | python3 -c "import sys,json; print(len(json.load(sys.stdin)))" 2>/dev/null || echo "0")
[ "$COUNT" = "0" ] && echo "No issues." && exit 0

echo "Found $COUNT issue(s). Starting parallel workers..."
for row in $(echo "$ISSUES" | python3 -c "import sys,json; [print(f\"{i['repo']}|{i['number']}|{i['title']}|{i['body'][:500]}\") for i in json.load(sys.stdin)]" 2>/dev/null); do
  IFS='|' read -r REPO NUM TITLE BODY <<< "$row"
  DESC="$TITLE"; [ -n "$BODY" ] && DESC="$TITLE: $BODY"
  (bash "$(dirname "$0")/dev-run-fix.sh" "$REPO" "$DESC" 2>&1) &
done
wait
echo "=== All workers finished ==="
```
